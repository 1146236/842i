def validate_az_license_plate(plate):
    if len(plate) != 9:
        return False
    if not plate[:2].isdigit():
        return False
    if plate[2] != "-":
        return False
    if not plate[3:5].isalpha() or not plate[3:5].isupper():
        return False
    if plate[5] != "-":
        return False
    if not plate[6:].isdigit():
        return False
    return True
 
print(validate_az_license_plate())
