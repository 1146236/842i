from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.common.exceptions import NoSuchElementException
from csv import writer
from time import sleep

driver = webdriver.Chrome('C:/Users/Admin/OneDrive/Desktop/driver')

driver.get('https://kontakt.az/telefonlar/mobil-telefonlar/samsung-mobil-telefonlar/')
with open('samsung.csv', 'w') as samsung_csv:
    csv_writer = writer(samsung_csv)
    csv_writer.writerow(['Description ', ' Price'])

    
    phones = driver.find_elements(By.CLASS_NAME, 'cart-body-top')
    for phone in phones:
        description = phone.find_element(By.CLASS_NAME, 'name').text
        price = phone.find_element(By.CLASS_NAME, 'nprice').text

        csv_writer.writerow([description, price])
    try:
        next_button = driver.find_element(By.CLASS_NAME, 'nextpostslink')
        next_button.click()
        sleep(5)
    except NoSuchElementException:
        driver.quit()
