from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.common.exceptions import NoSuchElementException
from csv import writer
from time import sleep

driver = webdriver.Chrome('C:/Users/Admin/OneDrive/Desktop/driver')

driver.get('https://bazarstore.az/search?controller=search&s=finish')

with open('store.csv', 'w', encoding='utf-8') as store_csv:
    csv_writer = writer(store_csv)
    csv_writer.writerow(['Title'])
    csv_writer.writerow(['Price'])

    while True:
        products = driver.find_elements(By.CLASS_NAME, 'bs-product-row')   
        for product in products: 
            title = product.find_element(By.CLASS_NAME, 'product-title').text
            price = product.find_element(By.CLASS_NAME, 'product-price-and-shipping').text

            csv_writer.writerow([title])
            csv_writer.writerow([price])
        try:
            next_button = driver.find_element(By.LINK_TEXT, 'Növbəti')
            next_button.click()
            sleep(5)
        except NoSuchElementException:
            driver.quit()
